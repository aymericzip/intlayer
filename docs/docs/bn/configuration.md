# Intlayer কনফিগারেশন ডকুমেন্টেশন

## ওভারভিউ

Intlayer কনফিগারেশন ফাইলগুলি আপনাকে প্লাগইনের বিভিন্ন দিক কাস্টমাইজ করতে দেয়, যেমন আন্তর্জাতিকীকরণ (i18n), মিডলওয়্যার এবং কন্টেন্ট ম্যানেজমেন্ট। এই ডকুমেন্টটি কনফিগারেশনের প্রতিটি প্রপার্টির বিস্তারিত বর্ণনা প্রদান করে।

## কনফিগারেশন ফাইল সমর্থন

Intlayer JSON, JS, MJS এবং TS কনফিগারেশন ফাইল ফরম্যাট গ্রহণ করে:

- `intlayer.config.ts`
- `intlayer.config.js`
- `intlayer.config.json`
- `intlayer.config.json5`
- `intlayer.config.jsonc`
- `intlayer.config.cjs`
- `intlayer.config.mjs`
- `.intlayerrc`

## কনফিগারেশন ফাইলের উদাহরণ

````typescript fileName="intlayer.config.ts" codeFormat="typescript"
import { Locales, type IntlayerConfig } from "intlayer";
import { nextjsRewrite } from "intlayer/routing";
import { z } from "zod";

/**
 * সকল উপলব্ধ বিকল্প সহ Intlayer কনফিগারেশন ফাইলের উদাহরণ।
 */
const config: IntlayerConfig = {
  /**
   * আন্তর্জাতিকীকরণ সেটিংস কনফিগারেশন।
   */
  internationalization: {
    /**
     * অ্যাপ্লিকেশনে সমর্থিত লোকালের তালিকা।
     * ডিফল্ট: [Locales.ENGLISH]
     */
    locales: [Locales.ENGLISH, Locales.FRENCH, Locales.SPANISH],

    /**
     * প্রয়োজনীয় লোকালের তালিকা যা প্রতিটি ডিকশনারিতে সংজ্ঞায়িত করা আবশ্যক।
     * যদি খালি থাকে, তবে `strict` মোডে সকল লোকাল প্রয়োজনীয়।
     * ডিফল্ট: []
     */
    requiredLocales: [Locales.ENGLISH],

    /**
     * আন্তর্জাতিকীকৃত কন্টেন্টের জন্য কঠোরতার স্তর।
     * - "strict": কোনো ঘোষিত লোকাল অনুপস্থিত থাকলে বা ঘোষিত না থাকলে ত্রুটি প্রদর্শন করে।
     * - "inclusive": কোনো ঘোষিত লোকাল অনুপস্থিত থাকলে সতর্কতা প্রদান করে।
     * - "loose": যেকোনো বিদ্যমান লোকাল গ্রহণ করে।
     * ডিফল্ট: "inclusive"
     */
    strictMode: "inclusive",

    /**
     * অনুরোধকৃত লোকাল না পাওয়া গেলে ফলব্যাক হিসেবে ব্যবহৃত ডিফল্ট লোকাল।
     * ডিফল্ট: Locales.ENGLISH
     */
    defaultLocale: Locales.ENGLISH,
  },

  /**
   * ডিকশনারি অপারেশন এবং অনুপস্থিত কন্টেন্টের আচরণ নিয়ন্ত্রণকারী সেটিংস।
   */
  dictionary: {
    /**
     * ডিকশনারি কীভাবে ইম্পোর্ট করা হয় তা নিয়ন্ত্রণ করে।
     * - "static": বিল্ডের সময় স্ট্যাটিক ইম্পোর্ট।
     * - "dynamic": Suspense ব্যবহার করে ডাইনামিক ইম্পোর্ট।
     * - "fetch": লাইভ সিঙ্ক API এর মাধ্যমে ডাইনামিক ফেচিং।
     * ডিফল্ট: "static"
     */
    importMode: "static",

    /**
     * AI ব্যবহার করে অনুপস্থিত অনুবাদগুলি স্বয়ংক্রিয়ভাবে পূরণ করার কৌশল।
     * একটি বুলিয়ান বা পূরণকৃত কন্টেন্ট সংরক্ষণের জন্য একটি পাথ প্যাটার্ন হতে পারে।
     * ডিফল্ট: true
     */
    fill: true,

    /**
     * ডিকশনারি ফাইলের ভৌত অবস্থান।
     * - "local": স্থানীয় ফাইল সিস্টেমে সংরক্ষিত।
     * - "remote": Intlayer CMS-এ সংরক্ষিত।
     * - "hybrid": স্থানীয় এবং Intlayer CMS উভয় স্থানেই সংরক্ষিত।
     * - "plugin" (বা যেকোনো কাস্টম স্ট্রিং): কোনো প্লাগইন বা কাস্টম সোর্স দ্বারা সরবরাহকৃত।
     * ডিফল্ট: "local"
     */
    location: "local",

    /**
     * কন্টেন্ট স্বয়ংক্রিয়ভাবে ট্রান্সফর্ম করা হবে কি না (যেমন মার্কডাউন থেকে HTML)।
     * ডিফল্ট: false
     */
    contentAutoTransformation: false,
  },

  /**
   * রাউটিং এবং মিডলওয়্যার কনফিগারেশন।
   */
  routing: {
    /**
     * প্রতি-লোকাল রাউটিং কৌশল।
     * - "prefix-no-default": ডিফল্ট ব্যতীত সব লোকালের জন্য প্রিফিক্স (যেমন /dashboard, /fr/dashboard)।
     * - "prefix-all": সব লোকালের জন্য প্রিফিক্স (যেমন /en/dashboard, /fr/dashboard)।
     * - "no-prefix": URL-এ কোনো লোকাল থাকবে না।
     * - "search-params": ?locale=... ব্যবহার করুন
     * ডিফল্ট: "prefix-no-default"
     */
    mode: "prefix-no-default",

    /**
     * ব্যবহারকারীর নির্বাচিত লোকাল কোথায় সংরক্ষণ করা হবে।
     * বিকল্পসমূহ: 'cookie', 'localStorage', 'sessionStorage', 'header' অথবা এগুলোর অ্যারে।
     * ডিফল্ট: ['cookie', 'header']
     */
    storage: ["cookie", "header"],

    /**
     * অ্যাপ্লিকেশন URL-এর বেস পাথ।
     * ডিফল্ট: ""
     */
    basePath: "",

    /**
     * নির্দিষ্ট লোকালের পাথের জন্য কাস্টম URL রিরাইট নিয়ম।
     */
    rewrite: nextjsRewrite({
      "/[locale]/about": {
        en: "/[locale]/about",
        fr: "/[locale]/a-propos",
      },
    }),

    /**
     * ডোমেইন-ভিত্তিক রাউটিং এর জন্য লোকালগুলিকে ডোমেইন হোস্টনামের সাথে ম্যাপ করে।
     * এই লোকালগুলির জন্য URL গুলি পরম (যেমন https://intlayer.cn/) হবে।
     * ডোমেইনটি লোকালকে নির্দেশ করে, তাই পাথে কোনো লোকাল প্রিফিক্স যোগ করা হয় না।
     * ডিফল্ট: undefined
     */
    domains: {
      en: "intlayer.org",
      zh: "intlayer.cn",
    },
  },

  /**
   * কন্টেন্ট ফাইল অনুসন্ধান এবং প্রক্রিয়াকরণের সেটিংস।
   */
  content: {
    /**
     * ডিকশনারি স্ক্যান করার জন্য ফাইল এক্সটেনশন।
     * ডিফল্ট: ['.content.ts', '.content.js', '.content.json', ইত্যাদি]
     */
    fileExtensions: [".content.ts", ".content.js", ".content.json"],

    /**
     * ডাইরেক্টরি যেখানে .content ফাইলগুলি অবস্থিত।
     * ডিফল্ট: ["."]
     */
    contentDir: ["src"],

    /**
     * সোর্স কোড ডাইরেক্টরি।
     * বিল্ড অপ্টিমাইজেশন এবং কোড ট্রান্সফরমেশনের জন্য ব্যবহৃত হয়।
     * ডিফল্ট: ["."]
     */
    codeDir: ["src"],

    /**
     * স্ক্যানিং থেকে বাদ দেওয়া প্যাটার্ন।
     * ডিফল্ট: ['node_modules', '.intlayer', ইত্যাদি]
     */
    excludedPath: ["node_modules"],

    /**
     * ডেভেলপমেন্টের সময় পরিবর্তনগুলি ট্র্যাক করা এবং ডিকশনারি পুনরায় তৈরি করা হবে কি না।
     * ডিফল্ট: ডেভেলপমেন্ট মোডে true
     */
    watch: true,

    /**
     * নতুন তৈরি / আপডেট করা .content ফাইলগুলি ফরম্যাট করার কমান্ড।
     */
    formatCommand: 'npx prettier --write "{{file}}"',
  },

  /**
   * ভিজ্যুয়াল এডিটর কনফিগারেশন।
   */
  editor: {
    /**
     * ভিজ্যুয়াল এডিটর সক্রিয় কি না।
     * ডিফল্ট: false
     */
    enabled: true,

### রাউটিং কনফিগারেশন (Routing)

URL কাঠামো, লোকাল স্টোরেজ এবং মিডলওয়্যার ম্যানেজমেন্ট সহ রাউটিং আচরণ নিয়ন্ত্রণকারী সেটিংস।

| ফিল্ড      | বর্ণনা                                                                                                                                                                                                          | টাইপ                                                                                                                                                                                                         | ডিফল্ট                 | উদাহরণ                                                                                                                                                                                            | মন্তব্য                                                                                                                                                                                                                                                                              |
| ---------- | --------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------ | ---------------------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------ |
| `mode`     | লোকাল ম্যানেজমেন্টের জন্য URL রাউটিং মোড।                                                                                                                                                                       | `'prefix-no-default'` &#124; <br/> `'prefix-all'` &#124; <br/> `'no-prefix'` &#124; <br/> `'search-params'`                                                                                                  | `'prefix-no-default'`  | `'prefix-no-default'`: `/dashboard` (en) অথবা `/fr/dashboard` (fr)। `'prefix-all'`: `/en/dashboard` । `'no-prefix'`: লোকাল অন্য উপায়ে ম্যানেজ করা হয়। `'search-params'`: `/dashboard?locale=fr` | এটি কুকি ম্যানেজমেন্ট বা লোকাল স্টোরেজকে প্রভাবিত করে না।                                                                                                                                                                                                                            |
| `storage`  | ক্লায়েন্টে লোকাল স্টোরেজের কনফিগারেশন।                                                                                                                                                                         | `false` &#124; <br/> `'cookie'` &#124; <br/> `'localStorage'` &#124; <br/> `'sessionStorage'` &#124; <br/> `'header'` &#124; <br/> `CookiesAttributes` &#124; <br/> `StorageAttributes` &#124; <br/> `Array` | `['cookie', 'header']` | `'localStorage'` <br/> `[{ type: 'cookie', name: 'custom-locale', secure: true }]`                                                                                                                | নিচে স্টোরেজ প্যারামিটার টেবিল দেখুন।                                                                                                                                                                                                                                                |
| `basePath` | অ্যাপ্লিকেশন URL-এর জন্য বেস পাথ।                                                                                                                                                                               | `string`                                                                                                                                                                                                     | `''`                   | `'/my-app'`                                                                                                                                                                                       | যদি অ্যাপ্লিকেশনটি `https://example.com/my-app` ঠিকানায় চলে, তবে basePath হলো `'/my-app'` এবং URL গুলো হয় `https://example.com/my-app/en` ।                                                                                                                                        |
| `rewrite`  | কাস্টম URL রিরাইট নিয়ম যা নির্দিষ্ট পাথের জন্য ডিফল্ট রাউটিং মোডকে ওভাররাইড করে। ডাইনামিক প্যারামিটার `[param]` সমর্থন করে।                                                                                    | `Record<string, StrictModeLocaleMap<string>>`                                                                                                                                                                | `undefined`            | উদাহরণ নিচে দেখুন                                                                                                                                                                                 | • রিরাইট নিয়মের অগ্রাধিকার `mode`-এর চেয়ে বেশি।<br/>• Next.js এবং Vite-এর সাথে কাজ করে।<br/>• `getLocalizedUrl()` স্বয়ংক্রিয়ভাবে যথাযথ নিয়ম প্রয়োগ করে।<br/>• [কাস্টম URL রিরাইট](https://github.com/aymericzip/intlayer/blob/main/docs/docs/en/custom_url_rewrites.md) দেখুন। |
| `domains`  | ডোমেইন-ভিত্তিক রাউটিং এর জন্য লোকালগুলিকে ডোমেইন হোস্টনামের সাথে ম্যাপ করে। সেট করা হলে, সেই লোকালের জন্য URL গুলি এই ডোমেইনটিকে বেস (পরম URL) হিসেবে ব্যবহার করে এবং পাথে কোনো লোকাল প্রিফিক্স যোগ করা হয় না। | `Partial<Record<Locale, string>>`                                                                                                                                                                            | `undefined`            | `{ zh: 'intlayer.zh', fr: 'intlayer.org' }`                                                                                                                                                       | • হোস্টনামে অন্তর্ভুক্ত না থাকলে ডিফল্ট প্রোটোকল হলো `https://` ।<br/>• ডোমেইন নিজেই লোকালকে শনাক্ত করে, তাই `/zh/` প্রিফিক্স যোগ করা হয় না।<br/>• `getLocalizedUrl('/', 'zh')` রিটার্ন করে `https://intlayer.zh/` ।                                                                |

**`rewrite` উদাহরণ**:

```typescript
routing: {
  mode: "prefix-no-default", // ফলব্যাক কৌশল
  rewrite: nextjsRewrite({
    "/about": {
      en: "/about",
      fr: "/a-propos",
    },
    "/product/[slug]": {
      en: "/product/[slug]",
      fr: "/produit/[slug]",
    },
    "/blog/[category]/[id]": {
      en: "/blog/[category]/[id]",
      fr: "/journal/[category]/[id]",
    },
  }),
}
```

#### স্টোরেজ প্যারামিটার (Storage)

| মান                | মন্তব্য                                                                                                                                                                                                   | বর্ণনা                                                                             |
| ------------------ | --------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------- | ---------------------------------------------------------------------------------- |
| `'cookie'`         | • GDPR বাস্তবায়নের জন্য যথাযথ ব্যবহারকারীর সম্মতি নিশ্চিত করুন।<br/>• `CookiesAttributes` (`{ type: 'cookie', name: 'custom-locale', secure: true, httpOnly: false }`) এর মাধ্যমে কনফিগারযোগ্য।          | লোকাল কুকিতে সংরক্ষণ করে — ক্লায়েন্ট এবং সার্ভার উভয় স্থান থেকেই অ্যাক্সেসযোগ্য। |
| `'localStorage'`   | • স্পষ্টভাবে মুছে না ফেলা পর্যন্ত এটি শেষ হয় না।<br/>• Intlayer Proxy এটি অ্যাক্সেস করতে পারে না।<br/>• `StorageAttributes` (`{ type: 'localStorage', name: 'custom-locale' }`) এর মাধ্যমে কনফিগারযোগ্য। | কোনো সময়সীমা ছাড়াই ব্রাউজারে লোকাল সংরক্ষণ করে — কেবল ক্লায়েন্ট-সাইড।           |
| `'sessionStorage'` | • ট্যাব/উইন্ডো বন্ধ করলে মুছে যায়।<br/>• Intlayer Proxy এটি অ্যাক্সেস করতে পারে না।<br/>• `StorageAttributes` (`{ type: 'sessionStorage', name: 'custom-locale' }`) এর মাধ্যমে কনফিগারযোগ্য।             | পেজ সেশনের মেয়াদের জন্য লোকাল সংরক্ষণ করে — কেবল ক্লায়েন্ট-সাইড।                 |
| `'header'`         | • API কলের জন্য দরকারী।<br/>• ক্লায়েন্ট-সাইড এটি অ্যাক্সেস করতে পারে না।<br/>• `StorageAttributes` (`{ type: 'header', name: 'custom-locale' }`) এর মাধ্যমে কনফিগারযোগ্য।                                | HTTP হেডারের মাধ্যমে লোকাল সংরক্ষণ বা পাস করে — কেবল সার্ভার-সাইড।                 |

#### কুকি বৈশিষ্ট্য (Cookies Attributes)

কুকিতে স্টোরেজ ব্যবহার করার সময় অতিরিক্ত বৈশিষ্ট্য সেট করা যেতে পারে:

| ফিল্ড      | বর্ণনা                                              | টাইপ                                                  |
| ---------- | --------------------------------------------------- | ----------------------------------------------------- |
| `name`     | কুকির নাম। ডিফল্ট: `'INTLAYER_LOCALE'`              | `string`                                              |
| `domain`   | কুকি ডোমেইন। ডিফল্ট: `undefined`                    | `string`                                              |
| `path`     | কুকি পাথ। ডিফল্ট: `undefined`                       | `string`                                              |
| `secure`   | HTTPS প্রয়োজন। ডিফল্ট: `undefined`                 | `boolean`                                             |
| `httpOnly` | HTTP-only ফ্ল্যাগ। ডিফল্ট: `undefined`              | `boolean`                                             |
| `sameSite` | SameSite পলিসি।                                     | `'strict'` &#124; <br/> `'lax'` &#124; <br/> `'none'` |
| `expires`  | মেয়াদের তারিখ বা দিনের সংখ্যা। ডিফল্ট: `undefined` | `Date` &#124; <br/> `number`                          |

#### কনফিগারেশন উদাহরণ

নতুন v7 রাউটিং কাঠামোর জন্য এখানে কিছু সাধারণ কনফিগারেশন উদাহরণ দেওয়া হলো:

**বেসিক কনফিগারেশন (ডিফল্ট)**:

```typescript
import { Locales, type IntlayerConfig } from "intlayer";
// intlayer.config.ts
const config: IntlayerConfig = {
  internationalization: {
    locales: ["en", "fr", "es"],
    defaultLocale: "en",
  },
  routing: {
    mode: "prefix-no-default",
    storage: "localStorage",
    basePath: "",
  },
};

export default config;
```

**GDPR সম্মতির সাথে কনফিগারেশন**:

```typescript
import { Locales, type IntlayerConfig } from "intlayer";
// intlayer.config.ts
const config: IntlayerConfig = {
  internationalization: {
    locales: ["en", "fr", "es"],
    defaultLocale: "en",
  },
  routing: {
    mode: "prefix-no-default",
    storage: [
      {
        type: "localStorage",
        name: "user-locale",
      },
      {
        type: "cookie",
        name: "user-locale",
        secure: true,
        sameSite: "strict",
        httpOnly: false,
      },
    ],
    basePath: "",
  },
};

export default config;
```

### কন্টেন্ট কনফিগারেশন (Content)

### AI কনফিগারেশন (AI)

প্রোভাইডার, মডেল এবং API কী সহ Intlayer-এর AI বৈশিষ্ট্য নিয়ন্ত্রণকারী সেটিংস।

আপনি যদি অ্যাক্সেস কী সহ [Intlayer Dashboard](https://app.intlayer.org/project)-এ রেজিস্টার করেন তবে এই কনফিগারেশনটি ঐচ্ছিক। Intlayer স্বয়ংক্রিয়ভাবে আপনার প্রয়োজন অনুযায়ী সবচেয়ে সাশ্রয়ী এবং দক্ষ AI সমাধান ম্যানেজ করবে। ডিফল্ট বিকল্পগুলো ব্যবহার করা দীর্ঘমেয়াদী সবচেয়ে ভালো সমর্থনের নিশ্চয়তা দেয় কারণ Intlayer-কে ক্রমাগত আপডেট করা হয় লেটেস্ট মডেলগুলো ব্যবহারের জন্য।

আপনি যদি আপনার নিজস্ব API কী বা কোনো নির্দিষ্ট মডেল ব্যবহার করতে পছন্দ করেন, তবে আপনি নিজের AI কনফিগারেশন সংজ্ঞায়িত করতে পারেন।
এই AI কনফিগারেশনটি আপনার Intlayer এনভায়রনমেন্টে গ্লোবালি ব্যবহৃত হবে। CLI কমান্ডগুলো `fill` এর মতো কমান্ডের জন্য ডিফল্টভাবে এই সেটিংস ব্যবহার করবে, সেইসাথে SDK, ভিজ্যুয়াল এডিটর এবং CMS-ও। আপনি কমান্ড প্যারামিটারের মাধ্যমে নির্দিষ্ট ক্ষেত্রে এই ডিফল্ট মানগুলো ওভাররাইড করতে পারেন।

Intlayer সর্বোচ্চ নমনীয়তার জন্য একাধিক AI প্রোভাইডার সমর্থন করে। বর্তমানে সমর্থিত প্রোভাইডারগুলো হলো:

### বিল্ড কনফিগারেশন (Build)

Intlayer কীভাবে আপনার অ্যাপ্লিকেশনের আন্তর্জাতিকীকরণকে অপ্টিমাইজ এবং কম্পাইল করে তা নিয়ন্ত্রণকারী প্যারামিটারসমূহ।

বিল্ড অপশনগুলো `@intlayer/babel` এবং `@intlayer/swc` প্লাগইনের ওপর প্রয়োগ করা হয়।

> ডেভেলপমেন্ট মোডে, Intlayer ডেভেলপমেন্ট প্রক্রিয়া সহজ করতে ডিকশনারির স্ট্যাটিক ইম্পোর্ট ব্যবহার করে।

> অপ্টিমাইজেশন চলাকালীন, Intlayer কোড স্প্লিটিং (chunking) অপ্টিমাইজ করার জন্য ডিকশনারি কলগুলো প্রতিস্থাপন করবে যাতে ফলাফল বান্ডেলটি কেবল সেই ডিকশনারিগুলোই ইম্পোর্ট করে যা বাস্তবে ব্যবহৃত হয়।
